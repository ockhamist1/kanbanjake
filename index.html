// --- MORTGAGE RATE LOGIC --- function initRateCharts() { const commonOptions = { responsive: true, maintainAspectRatio: false, interaction: { mode: 'index', intersect: false, }, plugins: { legend: { position: 'bottom', labels: { color: '#888', font: { size: 10, weight: 'bold' }, usePointStyle: true } }, tooltip: { backgroundColor: 'rgba(30, 30, 30, 0.9)', padding: 12, callbacks: { label: function(context) { const d = context.raw; if (!d) return ''; return [ ` ${context.dataset.label}: ${Number(d.y).toFixed(3)}%`, ` APR: ${Number(d.apr).toFixed(3)}%`, ` Points: ${Number(d.points).toFixed(3)}` ]; } } } }, scales: { y: { // This ensures the chart doesn't clip high rates beginAtZero: false, grace: '10%', grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#888', callback: v => v.toFixed(2) + '%' } }, x: { grid: { display: false }, ticks: { color: '#888', font: { size: 10 } } } } }; convChart = new Chart(document.getElementById('convChart'), { type: 'line', data: { datasets: [] }, options: commonOptions }); vaChart = new Chart(document.getElementById('vaChart'), { type: 'line', data: { datasets: [] }, options: commonOptions }); } function updateChart(chart, allData, productType) { const lenders = ["PenFed", "NFCU", "Rocket", "USAA"]; const colors = { "PenFed": "gray", "NFCU": "#22c55e", "Rocket": "#ef4444", "USAA": "#12395B" }; // 1. Filter and Sort Data const productData = allData .filter(d => d.product === productType) .sort((a, b) => new Date(a.date) - new Date(b.date)); // 2. Create standardized labels (dates) const labels = [...new Set(productData.map(d => d.date))]; chart.data.labels = labels; chart.data.datasets = lenders.map((lender) => ({ label: lender, data: labels.map(date => { const e = productData.find(d => d.date === date && d.lender === lender); // Force values to Numbers to prevent Chart.js string errors return e ? { x: date, y: parseFloat(e.rate), apr: parseFloat(e.apr), points: parseFloat(e.points) } : null; }), borderColor: colors[lender], backgroundColor: colors[lender], tension: 0.3, borderWidth: 2, pointRadius: 4, spanGaps: true // This connects the lines even if a day is missing })); chart.update(); }
